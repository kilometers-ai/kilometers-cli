1use super::{FilterDecision, ProxyContext, ProxyFilter};
2use crate::auth::JwtToken;
3use anyhow::{Context, Result};
4use async_trait::async_trait;
5use chrono::{DateTime, Utc};
6use serde::{Deserialize, Serialize};
7use std::collections::HashMap;
8use uuid::Uuid;

10#[derive(Debug, Clone)]
11pub struct EventSenderFilter {
12    api_endpoint: String,
13    client: reqwest::Client,
14    jwt_token: JwtToken,
15}
16
17#[derive(Debug, Serialize)]
18struct TelemetryEvent {
19    event_type: String,
20    timestamp: DateTime<Utc>,
21    user_id: Option<String>,
22    user_tier: String,
23    command: String,
24    args: Vec<String>,
25    session_id: String,
26    metadata: HashMap<String, serde_json::Value>,
27}
28
29#[derive(Debug, Deserialize)]
30struct TelemetryResponse {
31    status: String,
32    #[allow(dead_code)]
33    message: Option<String>,
34    events_remaining: Option<u64>,
35}
36
37impl EventSenderFilter {
38    pub fn new(api_endpoint: String, jwt_token: JwtToken) -> Self {
39        Self {
40            api_endpoint,
41            client: reqwest::Client::new(),
42            jwt_token,
43        }
44    }
45
46    async fn send_telemetry_event(&self, ctx: &ProxyContext) -> Result<()> {
47        let session_id = Uuid::new_v4().to_string();
48
49        let event = TelemetryEvent {
50            event_type: "command_execution".to_string(),
51            timestamp: Utc::now(),
52            user_id: self.jwt_token.claims.user_id.clone(),
53            user_tier: self
54                .jwt_token
55                .claims
56                .tier
57                .as_deref()
58                .unwrap_or("free")
59                .to_string(),
60            command: ctx.request.command.clone(),
61            args: ctx.request.args.clone(),
62            session_id,
63            metadata: ctx
64                .request
65                .metadata
66                .iter()
67                .map(|(k, v)| (k.clone(), serde_json::Value::String(v.clone())))
68                .collect(),
69        };
70
71        let response = self
72            .client
73            .post(&self.api_endpoint)
74            .bearer_auth(&self.jwt_token.token)
75            .json(&event)
76            .send()
77            .await
78            .context("Failed to send telemetry event")?;
79
80        match response.status().as_u16() {
81            200..=299 => {
82                if let Ok(telemetry_response) = response.json::<TelemetryResponse>().await {
83                    tracing::info!(
84                        "Telemetry event sent successfully: {}",
85                        telemetry_response.status
86                    );
87                    if let Some(remaining) = telemetry_response.events_remaining {
88                        tracing::debug!("Events remaining this month: {}", remaining);
89                    }
90                } else {
91                    tracing::info!("Telemetry event sent successfully");
92                }
93                Ok(())
94            }
95            429 => {
96                tracing::warn!("Rate limit reached for telemetry events - continuing execution");
97                Ok(())
98            }
99            status => Err(anyhow::anyhow!("Telemetry failed with status {}", status)),
100        }
101    }
102}
103
104#[async_trait]
105impl ProxyFilter for EventSenderFilter {
106    async fn check(&self, ctx: &ProxyContext) -> Result<FilterDecision> {
107        match self.send_telemetry_event(ctx).await {
108            Ok(_) => {
109                tracing::debug!("Telemetry event sent for command: {}", ctx.request.command);
110            }
111            Err(e) => {
112                tracing::warn!(
113                    "Failed to send telemetry event: {} - continuing execution",
114                    e
115                );
116            }
117        }
118
119        Ok(FilterDecision::Allow)
120    }
121
122    fn is_blocking(&self) -> bool {
123        false
124    }
125
126    fn name(&self) -> &str {
127        "EventSender"
128    }
129}
