14pub async fn handle_init(
15    config_path: &PathBuf,
16    api_key: Option<String>,
17    api_url: String,
18) -> Result<()> {
19    if Config::exists(config_path) {
20        println!("Configuration already exists at {:?}", config_path);
21        print!("Overwrite? (y/N): ");
22        std::io::Write::flush(&mut std::io::stdout())?;
23
24        let mut input = String::new();
25        std::io::stdin().read_line(&mut input)?;
26        if !input.trim().eq_ignore_ascii_case("y") {
27            println!("Cancelled.");
28            return Ok(());
29        }
30    }
31
32    // Load .env file if it exists
33    if let Ok(dotenv_path) = std::env::current_dir().map(|mut p| {
34        p.push(".env");
35        p
36    }) {
37        if dotenv_path.exists() {
38            dotenvy::from_path(&dotenv_path).ok();
39        }
40    }
41
42    // Check for KM_API_URL environment variable (highest priority)
43    let api_url = std::env::var("KM_API_URL").unwrap_or(api_url);
44
45    let api_key = api_key
46        .or_else(|| {
47            print!("Enter your API key: ");
48            std::io::Write::flush(&mut std::io::stdout()).ok();
49            let mut input = String::new();
50            std::io::stdin().read_line(&mut input).ok();
51            Some(input.trim().to_string())
52        })
53        .context("API key is required")?;
54
55    // Validate API key by exchanging for JWT
56    println!("Validating API key...");
57    let auth_client = auth::AuthClient::new(api_key.clone(), api_url.clone());
58
59    match auth_client.exchange_for_jwt().await {
60        Ok(jwt_token) => {
61            println!("âœ“ Authentication successful");
62
63            // Store tokens in keyring
64            tracing::debug!(
65                "JWT token to save: token_len={}, expires_at={}, has_refresh={}",
66                jwt_token.token.len(),
67                jwt_token.expires_at,
68                jwt_token.refresh_token.is_some()
69            );
70            tracing::debug!(
71                "JWT claims: user_id={:?}, tier={:?}",
72                jwt_token.claims.user_id,
73                jwt_token.claims.tier
74            );
75
76            match KeyringTokenStore::new() {
77                Ok(token_store) => {
78                    let refresh_token = jwt_token.refresh_token.as_deref();
79
80                    // Try to serialize to see what will be saved
81                    match serde_json::to_string(&jwt_token) {
82                        Ok(serialized) => {
83                            tracing::debug!("Serialized JWT token: {}", serialized);
84                        }
85                        Err(e) => {
86                            tracing::error!("Failed to serialize JWT token: {}", e);
87                        }
88                    }
89
90                    if let Err(e) = token_store.save_tokens(&jwt_token, refresh_token) {
91                        tracing::error!("Failed to save tokens to keyring: {}", e);
92                        println!("âš  Warning: Could not save tokens to keyring: {}", e);
93                        println!("  Tokens will be refreshed on first use");
94                    } else {
95                        tracing::info!("Successfully saved tokens to keyring");
96                        println!("âœ“ Tokens stored securely in keyring");
97
98                        // Immediately verify we can read it back
99                        match token_store.load_access_token() {
100                            Ok(loaded_token) => {
101                                tracing::debug!("Verified token can be loaded back from keyring");
102                                tracing::debug!(
103                                    "Loaded token: user_id={:?}, tier={:?}",
104                                    loaded_token.claims.user_id,
105                                    loaded_token.claims.tier
106                                );
107                            }
108                            Err(e) => {
109                                tracing::error!("Failed to load token back from keyring: {}", e);
110                                println!("âš  Warning: Token was saved but cannot be loaded: {}", e);
111                            }
112                        }
113                    }
114                }
115                Err(e) => {
116                    tracing::error!("Failed to initialize keyring: {}", e);
117                    println!("âš  Warning: Could not initialize keyring: {}", e);
118                    println!("  Tokens will be refreshed on first use");
119                }
120            }
121
122            // Save config only after successful authentication
123            let config = Config::new(api_key, api_url);
124            config.save(config_path)?;
125            println!("âœ“ Configuration saved to {:?}", config_path);
126
127            // Display user info if available
128            if let Some(user_id) = &jwt_token.claims.user_id {
129                println!("âœ“ Authenticated as user: {}", user_id);
130            }
131            if let Some(tier) = &jwt_token.claims.tier {
132                println!("âœ“ User tier: {}", tier);
133            }

135            Ok(())
136        }
137        Err(e) => {
138            println!("âœ— Authentication failed: {}", e);

140            println!("  â€¢ Your API key is correct");
141            println!("  â€¢ You have network connectivity");
142            println!("  â€¢ The API URL is correct: {}", api_url);
143            Err(anyhow::anyhow!(
144                "Failed to authenticate with provided API key"
145            ))
146        }

150pub async fn get_jwt_token_with_cache(api_key: String, api_url: String) -> Option<JwtToken> {
151    let token_store = match KeyringTokenStore::new() {
152        Ok(store) => store,
153        Err(e) => {
154            tracing::warn!("Failed to initialize keyring token store: {}", e);
155            return None;
156        }
157    };
158
159    // Try to load from keyring first
160    if token_store.token_exists() {
161        if let Ok(cached_token) = token_store.load_access_token() {
162            if !auth::AuthClient::is_token_expired(&cached_token) {
163                tracing::debug!("Using cached JWT token from keyring");
164                return Some(cached_token);
165            } else {
166                tracing::debug!("Cached token expired, fetching new token");
167            }
168        } else {
169            tracing::debug!("Failed to load cached token from keyring, fetching new token");
170        }
171    } else {
172        tracing::debug!("No token found in keyring, fetching new token");
173    }
174
175    // Exchange for new token
176    let auth_client = auth::AuthClient::new(api_key, api_url);
177    match auth_client.exchange_for_jwt().await {
178        Ok(new_token) => {
179            // Save to keyring
180            let refresh_token = new_token.refresh_token.as_deref();
181            if let Err(e) = token_store.save_tokens(&new_token, refresh_token) {
182                tracing::warn!(
183                    "Failed to save token to keyring: {} - continuing without keyring",
184                    e
185                );
186            } else {
187                tracing::debug!("Token saved to keyring successfully");
188            }
189            Some(new_token)
190        }
191        Err(e) => {
192            tracing::warn!(
193                "Authentication failed: {} - continuing in local-only mode",
194                e
195            );
196            None
197        }
198    }
199}

201pub async fn handle_monitor(
202    config_path: &Path,
203    args: Vec<String>,
204    local_only: bool,
205    override_tier: Option<String>,
206    log_file: PathBuf,
207) -> Result<()> {
208    if args.is_empty() {
209        return Err(anyhow::anyhow!("No command provided to proxy"));
210    }
211
212    let program = args[0].clone();
213    let program_args = args[1..].to_vec();
214
215    tracing::info!("Proxying command: {} {:?}", program, program_args);
216
217    // Load config with environment variable support, but gracefully handle missing config
218    let default_api_url = "https://api.kilometers.ai".to_string();
219    let (jwt_token_option, api_url) = if local_only {
220        tracing::info!("Running in local-only mode - skipping authentication");
221        (None, default_api_url)
222    } else {
223        match Config::load_with_env(config_path) {
224            Ok(config) => {
225                let (api_key, api_url) = (config.api_key, config.api_url.clone());
226                let token = get_jwt_token_with_cache(api_key, api_url.clone()).await;
227                (token, api_url)
228            }
229            Err(e) => {
230                tracing::info!("No configuration found - running in local-only mode. Use 'km init' to set up cloud features.");
231                tracing::debug!("Config load error: {}", e);
232                (None, default_api_url)
233            }
234        }
235    };
236
237    let (user_tier, jwt_token) = if let Some(token) = jwt_token_option {
238        let tier = override_tier
239            .as_deref()
240            .or(token.claims.tier.as_deref())
241            .unwrap_or("free");
242        tracing::info!("User tier: {}", tier);
243        (tier.to_string(), Some(token))
244    } else {
245        tracing::info!("Authentication failed - running in local-only mode");
246        ("free".to_string(), None)
247    };
248
249    let proxy_request = ProxyRequest::new(program.clone(), program_args.clone());
250    let proxy_context = ProxyContext::new(
251        proxy_request,
252        jwt_token
253            .as_ref()
254            .map(|t| t.token.clone())
255            .unwrap_or_default(),
256    );
257
258    let pipeline = if local_only || jwt_token.is_none() {
259        if local_only {
260            tracing::info!("Using local logging only (--local-only specified)");
261        } else {
262            tracing::info!("Using local logging only (authentication failed)");
263        }
264        // Use separate log file for command metadata vs MCP traffic
265        let metadata_log = log_file
266            .parent()
267            .unwrap_or_else(|| std::path::Path::new("."))
268            .join("km_commands.log");
269        FilterPipeline::new().add_filter(Box::new(LocalLoggerFilter::new(metadata_log)))
270    } else if let Some(token) = jwt_token.clone() {
271        tracing::info!(
272            "Using filter pipeline with telemetry for {} tier",
273            user_tier
274        );
275        let mut pipeline = FilterPipeline::new()
276            .add_filter(Box::new(LocalLoggerFilter::new(log_file.clone())))
277            .add_filter(Box::new(EventSenderFilter::new(
278                format!("{}/api/events/telemetry", api_url),
279                token.clone(),
280            )));
281
282        if user_tier != "free" {
283            tracing::info!("Adding risk analysis for paid tier user");
284            pipeline = pipeline.add_filter(Box::new(RiskAnalysisFilter::new(
285                format!("{}/api/risk/analyze", api_url),
286                0.8,
287            )));
288        }
289
290        pipeline
291    } else {
292        // Fallback case (should not happen but be safe)
293        tracing::info!("Fallback to local logging only");
294        // Use separate log file for command metadata vs MCP traffic
295        let metadata_log = log_file
296            .parent()
297            .unwrap_or_else(|| std::path::Path::new("."))
298            .join("km_commands.log");
299        FilterPipeline::new().add_filter(Box::new(LocalLoggerFilter::new(metadata_log)))
300    };
301
302    match pipeline.execute(proxy_context).await {
303        Ok(filtered_request) => {
304            tracing::info!("Request approved, executing proxy");
305            proxy::run_proxy(&filtered_request.command, &filtered_request.args, &log_file)?;
306        }
307        Err(e) => {
308            return Err(anyhow::anyhow!("Request blocked: {}", e));
309        }
310    }
311
312    Ok(())
313}

315pub fn handle_clear_logs(include_config: bool, config_path: &Path) -> Result<()> {
316    let log_files = vec!["mcp_traffic.jsonl", "mcp_requests.log", "mcp_proxy.log"];
317    let mut had_errors = false;
318
319    for file in log_files {
320        let file_path = PathBuf::from(file);
321        if file_path.exists() {
322            match fs::remove_file(&file_path) {
323                Ok(_) => println!("âœ“ Deleted {}", file),
324                Err(e) => {
325                    tracing::warn!("Failed to delete {}: {}", file, e);
326                    had_errors = true;
327                    // Continue processing other files
328                }
329            }
330        }
331    }
332
333    if include_config && config_path.exists() {
334        match fs::remove_file(config_path) {
335            Ok(_) => println!("âœ“ Deleted config at {:?}", config_path),
336            Err(e) => {
337                tracing::warn!("Failed to delete config: {}", e);
338                had_errors = true;
339            }
340        }
341    }
342
343    // Clear tokens from keyring (best effort, don't fail if keyring is unavailable)
344    match KeyringTokenStore::new() {
345        Ok(token_store) => {
346            if token_store.token_exists() {
347                if let Err(e) = token_store.clear_tokens() {
348                    tracing::warn!("Failed to clear tokens from keyring: {}", e);
349                    // Don't mark as error - keyring might not be available in CI
350                } else {
351                    println!("âœ“ Cleared tokens from keyring");
352                }
353            }
354        }
355        Err(e) => {
356            // Keyring not available (e.g., in CI), just log and continue
357            tracing::debug!("Keyring not available: {}", e);
358        }
359    }
360
361    if had_errors {
362        println!("Logs cleared with some warnings (see above).");
363    } else {
364        println!("All logs cleared.");
365    }
366
367    Ok(())
368}

403pub fn handle_logs(
404    file: PathBuf,
405    requests: bool,
406    responses: bool,
407    method: Option<String>,
408    tail: bool,
409    lines: Option<usize>,
410) -> Result<()> {
411    if !file.exists() {
412        return Err(anyhow::anyhow!("Log file {:?} not found", file));
413    }
414
415    let contents = fs::read_to_string(&file)?;
416    let all_lines: Vec<&str> = contents.lines().collect();
417
418    let filtered_lines: Vec<&str> = all_lines
419        .iter()
420        .filter(|line| {
421            if let Ok(json) = serde_json::from_str::<serde_json::Value>(line) {
422                // Filter by direction
423                if requests && json.get("direction") == Some(&serde_json::json!("response")) {
424                    return false;
425                }
426                if responses && json.get("direction") == Some(&serde_json::json!("request")) {
427                    return false;
428                }
429
430                // Filter by method
431                if let Some(ref m) = method {
432                    if let Some(content) = json.get("content").and_then(|c| c.as_str()) {
433                        if let Ok(rpc) = serde_json::from_str::<serde_json::Value>(content) {
434                            if rpc.get("method").and_then(|v| v.as_str()) != Some(m) {
435                                return false;
436                            }
437                        }
438                    }
439                }
440
441                true
442            } else {
443                false
444            }
445        })
446        .copied()
447        .collect();
448
449    let display_lines = if let Some(n) = lines {
450        filtered_lines.iter().rev().take(n).rev().copied().collect()
451    } else {
452        filtered_lines
453    };
454
455    for line in display_lines {
456        if let Ok(json) = serde_json::from_str::<serde_json::Value>(line) {
457            println!("{}", serde_json::to_string_pretty(&json)?);
458        }
459    }
460
461    if tail {
462        println!("\nTail mode not yet implemented.");
463    }
464
465    Ok(())
466}

468pub fn handle_doctor_jwt() -> Result<()> {
469    println!("JWT Token Information:");
470    println!();
471
472    // Try to initialize keyring token store
473    let token_store = match KeyringTokenStore::new() {
474        Ok(store) => {
475            tracing::debug!("Keyring token store initialized successfully");
476            store
477        }
478        Err(e) => {
479            tracing::error!("Failed to initialize keyring: {}", e);
480            println!("âœ— Failed to initialize keyring: {}", e);
481            println!("\nThe keyring may not be available on this system.");
482            println!("Run 'km init' to configure authentication.");
483            return Ok(());
484        }
485    };
486
487    // Check if token exists
488    let exists = token_store.token_exists();
489    tracing::debug!("Token exists in keyring: {}", exists);
490
491    if !exists {
492        println!("âœ— No JWT token found in keyring");
493        println!("\nRun 'km init' to authenticate and store a token.");
494
495        // Try to get raw password to debug
496        tracing::debug!("Attempting to read raw keyring entry for debugging...");
497        return Ok(());
498    }
499
500    // Load the token
501    tracing::debug!("Attempting to load access token from keyring...");
502    match token_store.load_access_token() {
503        Ok(jwt_token) => {
504            // Display token (truncated for security)
505            let token_display = if jwt_token.token.len() > 40 {
506                format!(
507                    "{}...{}",
508                    &jwt_token.token[..20],
509                    &jwt_token.token[jwt_token.token.len() - 20..]
510                )
511            } else {
512                jwt_token.token.clone()
513            };
514            println!("  Token: {}", token_display);
515
516            // Display expiration
517            use std::time::{SystemTime, UNIX_EPOCH};
518            let now = SystemTime::now()
519                .duration_since(UNIX_EPOCH)
520                .unwrap()
521                .as_secs();
522
523            let is_expired = AuthClient::is_token_expired(&jwt_token);
524            let expires_in = jwt_token.expires_at.saturating_sub(now);
525
526            let expires_at_str = chrono::DateTime::from_timestamp(jwt_token.expires_at as i64, 0)
527                .map(|dt| dt.format("%Y-%m-%d %H:%M:%S UTC").to_string())
528                .unwrap_or_else(|| "Invalid timestamp".to_string());
529
530            println!(
531                "  Expires At: {} (in {} seconds)",
532                expires_at_str, expires_in
533            );
534
535            if is_expired {
536                println!("  Status: âœ— EXPIRED");
537            } else {
538                println!("  Status: âœ“ Valid");
539            }
540
541            println!();
542            println!("  Claims:");
543
544            if let Some(user_id) = &jwt_token.claims.user_id {
545                println!("    User ID: {}", user_id);
546            }
547
548            if let Some(tier) = &jwt_token.claims.tier {
549                println!("    Tier: {}", tier);
550            }
551
552            if let Some(sub) = &jwt_token.claims.sub {
553                println!("    Subject: {}", sub);
554            }
555
556            if let Some(iat) = jwt_token.claims.iat {
557                let iat_str = chrono::DateTime::from_timestamp(iat as i64, 0)
558                    .map(|dt| dt.format("%Y-%m-%d %H:%M:%S UTC").to_string())
559                    .unwrap_or_else(|| "Invalid timestamp".to_string());
560                println!("    Issued At: {}", iat_str);
561            }
562
563            println!();
564
565            if is_expired {
566                println!("âš  Token is expired. Run 'km monitor' to automatically refresh it,");
567                println!("  or run 'km init' to re-authenticate.");
568            }
569
570            Ok(())
571        }
572        Err(e) => {
573            tracing::error!("Failed to load JWT token from keyring: {}", e);
574            println!("âœ— Failed to load JWT token: {}", e);
575            println!("\nThe token may be corrupted. Run 'km init' to re-authenticate.");
576            Ok(())
577        }
578    }
