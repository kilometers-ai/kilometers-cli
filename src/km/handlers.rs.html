15pub async fn handle_init(
16    config_path: &PathBuf,
17    api_key: Option<String>,
18    api_url: String,
19) -> Result<()> {
20    if Config::exists(config_path) {
21        println!("Configuration already exists at {:?}", config_path);
22        print!("Overwrite? (y/N): ");
23        std::io::Write::flush(&mut std::io::stdout())?;
24
25        let mut input = String::new();
26        std::io::stdin().read_line(&mut input)?;
27        if !input.trim().eq_ignore_ascii_case("y") {
28            println!("Cancelled.");
29            return Ok(());
30        }
31    }
32
33    // Load .env file if it exists
34    if let Ok(dotenv_path) = std::env::current_dir().map(|mut p| {
35        p.push(".env");
36        p
37    }) {
38        if dotenv_path.exists() {
39            dotenvy::from_path(&dotenv_path).ok();
40        }
41    }
42
43    // Check for KM_API_URL environment variable (highest priority)
44    let api_url = std::env::var("KM_API_URL").unwrap_or(api_url);
45
46    // If no valid JWT in keyring, attempt device code flow first
47    let should_try_device_flow = match KeyringTokenStore::new() {
48        Ok(store) => {
49            if store.token_exists() {
50                match store.load_access_token() {
51                    Ok(tok) => auth::AuthClient::is_token_expired(&tok),
52                    Err(_) => true,
53                }
54            } else {
55                true
56            }
57        }
58        Err(_) => true,
59    };
60
61    if should_try_device_flow {
62        println!("No valid login found. Starting device sign-in...");
63        let dac = DeviceAuthClient::new(api_url.clone());
64        if let Ok(start) = dac.start().await {
65            println!("\nOpen to sign in: {}", start.verification_uri_complete);
66            println!(
67                "Or visit {} and enter code: {}\n",
68                start.verification_uri, start.user_code
69            );
70            println!("Device Code (DEBUG): {}", start.device_code);
71
72            // Best-effort browser open
73            #[cfg(target_os = "macos")]
74            let _ = std::process::Command::new("open")
75                .arg(&start.verification_uri_complete)
76                .status();
77            #[cfg(target_os = "windows")]
78            let _ = std::process::Command::new("cmd")
79                .args(["/C", "start", &start.verification_uri_complete])
80                .status();
81            #[cfg(target_os = "linux")]
82            let _ = std::process::Command::new("xdg-open")
83                .arg(&start.verification_uri_complete)
84                .status();
85
86            let deadline =
87                std::time::Instant::now() + std::time::Duration::from_secs(start.expires_in + 10);
88            let mut interval = std::time::Duration::from_secs(start.interval.max(5));
89            loop {
90                if std::time::Instant::now() > deadline {
91                    println!("Device code expired. Falling back to API key auth.");
92                    break;
93                }
94
95                match dac.poll(&start.device_code).await {
96                    Ok(Ok(success)) => {
97                        println!("✓ Sign-in approved");
98                        // Build JwtToken
99                        let claims =
100                            auth::AuthClient::parse_jwt_claims(&success.token.access_token)
101                                .unwrap_or_default();
102                        let now = std::time::SystemTime::now()
103                            .duration_since(std::time::UNIX_EPOCH)
104                            .unwrap()
105                            .as_secs();
106                        let jwt = JwtToken {
107                            token: success.token.access_token.clone(),
108                            expires_at: now + 3600, // dashboard returns RFC3339; we use 1h default if parsing not implemented here
109                            claims,
110                            refresh_token: None,
111                        };
112                        if let Ok(store) = KeyringTokenStore::new() {
113                            let _ = store.save_tokens(&jwt, None);
114                        }
115                        // Save config with empty API key (we're using JWT tokens now)
116                        let cfg = Config::new(String::new(), api_url.clone());
117                        cfg.save(config_path)?;
118                        println!("✓ Configuration saved to {:?}", config_path);
119                        return Ok(());
120                    }
121                    Ok(Err(err)) => match err.as_str() {
122                        "authorization_pending" => {
123                            tokio::time::sleep(interval).await;
124                        }
125                        "slow_down" => {
126                            interval += std::time::Duration::from_secs(5);
127                            tokio::time::sleep(interval).await;
128                        }
129                        "expired_token" | "access_denied" => {
130                            println!(
131                                "{}",
132                                if err == "expired_token" {
133                                    "Device code expired"
134                                } else {
135                                    "Access denied"
136                                }
137                            );
138                            break;
139                        }
140                        _ => {
141                            tokio::time::sleep(interval).await;
142                        }
143                    },
144                    Err(_) => {
145                        tokio::time::sleep(interval).await;
146                    }
147                }
148            }
149        }
150    }
151
152    let api_key = api_key
153        .or_else(|| {
154            print!("Enter your API key: ");
155            std::io::Write::flush(&mut std::io::stdout()).ok();
156            let mut input = String::new();
157            std::io::stdin().read_line(&mut input).ok();
158            Some(input.trim().to_string())
159        })
160        .context("API key is required")?;
161
162    // Validate API key by exchanging for JWT
163    println!("Validating API key...");
164    let auth_client = auth::AuthClient::new(api_key.clone(), api_url.clone());
165
166    match auth_client.exchange_for_jwt().await {
167        Ok(jwt_token) => {
168            println!("✓ Authentication successful");
169
170            // Store tokens in keyring
171            tracing::debug!(
172                "JWT token to save: token_len={}, expires_at={}, has_refresh={}",
173                jwt_token.token.len(),
174                jwt_token.expires_at,
175                jwt_token.refresh_token.is_some()
176            );
177            tracing::debug!(
178                "JWT claims: user_id={:?}, tier={:?}",
179                jwt_token.claims.user_id,
180                jwt_token.claims.tier
181            );
182
183            match KeyringTokenStore::new() {
184                Ok(token_store) => {
185                    let refresh_token = jwt_token.refresh_token.as_deref();
186
187                    // Try to serialize to see what will be saved
188                    match serde_json::to_string(&jwt_token) {
189                        Ok(serialized) => {
190                            tracing::debug!("Serialized JWT token: {}", serialized);
191                        }
192                        Err(e) => {
193                            tracing::error!("Failed to serialize JWT token: {}", e);
194                        }
195                    }
196
197                    if let Err(e) = token_store.save_tokens(&jwt_token, refresh_token) {
198                        tracing::error!("Failed to save tokens to keyring: {}", e);
199                        println!("⚠ Warning: Could not save tokens to keyring: {}", e);
200                        println!("  Tokens will be refreshed on first use");
201                    } else {
202                        tracing::info!("Successfully saved tokens to keyring");
203                        println!("✓ Tokens stored securely in keyring");
204
205                        // Immediately verify we can read it back
206                        match token_store.load_access_token() {
207                            Ok(loaded_token) => {
208                                tracing::debug!("Verified token can be loaded back from keyring");
209                                tracing::debug!(
210                                    "Loaded token: user_id={:?}, tier={:?}",
211                                    loaded_token.claims.user_id,
212                                    loaded_token.claims.tier
213                                );
214                            }
215                            Err(e) => {
216                                tracing::error!("Failed to load token back from keyring: {}", e);
217                                println!("⚠ Warning: Token was saved but cannot be loaded: {}", e);
218                            }
219                        }
220                    }
221                }
222                Err(e) => {
223                    tracing::error!("Failed to initialize keyring: {}", e);
224                    println!("⚠ Warning: Could not initialize keyring: {}", e);
225                    println!("  Tokens will be refreshed on first use");
226                }
227            }
228
229            // Save config only after successful authentication
230            let config = Config::new(api_key, api_url);
231            config.save(config_path)?;
232            println!("✓ Configuration saved to {:?}", config_path);
233
234            // Display user info if available
235            if let Some(user_id) = &jwt_token.claims.user_id {
236                println!("✓ Authenticated as user: {}", user_id);
237            }
238            if let Some(tier) = &jwt_token.claims.tier {
239                println!("✓ User tier: {}", tier);
240            }
241
242            Ok(())
243        }
244        Err(e) => {
245            println!("✗ Authentication failed: {}", e);
246            println!("\nPlease check:");
247            println!("  • Your API key is correct");
248            println!("  • You have network connectivity");
249            println!("  • The API URL is correct: {}", api_url);
250            Err(anyhow::anyhow!(
251                "Failed to authenticate with provided API key"
252            ))
253        }
254    }
255}

257pub async fn get_jwt_token_with_cache(api_key: String, api_url: String) -> Option<JwtToken> {
258    let token_store = match KeyringTokenStore::new() {
259        Ok(store) => store,
260        Err(e) => {
261            tracing::warn!("Failed to initialize keyring token store: {}", e);
262            return None;
263        }
264    };
265
266    // Try to load from keyring first
267    if token_store.token_exists() {
268        if let Ok(cached_token) = token_store.load_access_token() {
269            if !auth::AuthClient::is_token_expired(&cached_token) {
270                tracing::debug!("Using cached JWT token from keyring");
271                return Some(cached_token);
272            } else {
273                tracing::debug!("Cached token expired, fetching new token");
274            }
275        } else {
276            tracing::debug!("Failed to load cached token from keyring, fetching new token");
277        }
278    } else {
279        tracing::debug!("No token found in keyring, fetching new token");
280    }
281
282    // Exchange for new token
283    let auth_client = auth::AuthClient::new(api_key, api_url);
284    match auth_client.exchange_for_jwt().await {
285        Ok(new_token) => {
286            // Save to keyring
287            let refresh_token = new_token.refresh_token.as_deref();
288            if let Err(e) = token_store.save_tokens(&new_token, refresh_token) {
289                tracing::warn!(
290                    "Failed to save token to keyring: {} - continuing without keyring",
291                    e
292                );
293            } else {
294                tracing::debug!("Token saved to keyring successfully");
295            }
296            Some(new_token)
297        }
298        Err(e) => {
299            tracing::warn!(
300                "Authentication failed: {} - continuing in local-only mode",
301                e
302            );
303            None
304        }
305    }
306}

308pub async fn handle_monitor(
309    config_path: &Path,
310    args: Vec<String>,
311    local_only: bool,
312    override_tier: Option<String>,
313    log_file: PathBuf,
314) -> Result<()> {
315    if args.is_empty() {
316        return Err(anyhow::anyhow!("No command provided to proxy"));
317    }
318
319    let program = args[0].clone();
320    let program_args = args[1..].to_vec();
321
322    tracing::info!("Proxying command: {} {:?}", program, program_args);
323
324    // Load config with environment variable support, but gracefully handle missing config
325    let default_api_url = "https://api.kilometers.ai".to_string();
326    let (jwt_token_option, api_url) = if local_only {
327        tracing::info!("Running in local-only mode - skipping authentication");
328        (None, default_api_url)
329    } else {
330        match Config::load_with_env(config_path) {
331            Ok(config) => {
332                let (api_key, api_url) = (config.api_key, config.api_url.clone());
333                let token = get_jwt_token_with_cache(api_key, api_url.clone()).await;
334                (token, api_url)
335            }
336            Err(e) => {
337                tracing::info!("No configuration found - running in local-only mode. Use 'km init' to set up cloud features.");
338                tracing::debug!("Config load error: {}", e);
339                (None, default_api_url)
340            }
341        }
342    };
343
344    let (user_tier, jwt_token) = if let Some(token) = jwt_token_option {
345        let tier = override_tier
346            .as_deref()
347            .or(token.claims.tier.as_deref())
348            .unwrap_or("free");
349        tracing::info!("User tier: {}", tier);
350        (tier.to_string(), Some(token))
351    } else {
352        tracing::info!("Authentication failed - running in local-only mode");
353        ("free".to_string(), None)
354    };
355
356    let proxy_request = ProxyRequest::new(program.clone(), program_args.clone());
357    let proxy_context = ProxyContext::new(
358        proxy_request,
359        jwt_token
360            .as_ref()
361            .map(|t| t.token.clone())
362            .unwrap_or_default(),
363    );
364
365    let pipeline = if local_only || jwt_token.is_none() {
366        if local_only {
367            tracing::info!("Using local logging only (--local-only specified)");
368        } else {
369            tracing::info!("Using local logging only (authentication failed)");
370        }
371        // Use separate log file for command metadata vs MCP traffic
372        let metadata_log = log_file
373            .parent()
374            .unwrap_or_else(|| std::path::Path::new("."))
375            .join("km_commands.log");
376        FilterPipeline::new().add_filter(Box::new(LocalLoggerFilter::new(metadata_log)))
377    } else if let Some(token) = jwt_token.clone() {
378        tracing::info!(
379            "Using filter pipeline with telemetry for {} tier",
380            user_tier
381        );
382        let mut pipeline = FilterPipeline::new()
383            .add_filter(Box::new(LocalLoggerFilter::new(log_file.clone())))
384            .add_filter(Box::new(EventSenderFilter::new(
385                format!("{}/api/events/telemetry", api_url),
386                token.clone(),
387            )));
388
389        if user_tier != "free" {
390            tracing::info!("Adding risk analysis for paid tier user");
391            pipeline = pipeline.add_filter(Box::new(RiskAnalysisFilter::new(
392                format!("{}/api/risk/analyze", api_url),
393                0.8,
394            )));
395        }
396
397        pipeline
398    } else {
399        // Fallback case (should not happen but be safe)
400        tracing::info!("Fallback to local logging only");
401        // Use separate log file for command metadata vs MCP traffic
402        let metadata_log = log_file
403            .parent()
404            .unwrap_or_else(|| std::path::Path::new("."))
405            .join("km_commands.log");
406        FilterPipeline::new().add_filter(Box::new(LocalLoggerFilter::new(metadata_log)))
407    };
408
409    match pipeline.execute(proxy_context).await {
410        Ok(filtered_request) => {
411            tracing::info!("Request approved, executing proxy");
412            proxy::run_proxy(&filtered_request.command, &filtered_request.args, &log_file)?;
413        }
414        Err(e) => {
415            return Err(anyhow::anyhow!("Request blocked: {}", e));
416        }
417    }
418
419    Ok(())
420}

422pub fn handle_clear_logs(include_config: bool, config_path: &Path) -> Result<()> {
423    let log_files = vec!["mcp_traffic.jsonl", "mcp_requests.log", "mcp_proxy.log"];
424    let mut had_errors = false;
425
426    for file in log_files {
427        let file_path = PathBuf::from(file);
428        if file_path.exists() {
429            match fs::remove_file(&file_path) {
430                Ok(_) => println!("✓ Deleted {}", file),
431                Err(e) => {
432                    tracing::warn!("Failed to delete {}: {}", file, e);
433                    had_errors = true;
434                    // Continue processing other files
435                }
436            }
437        }
438    }
439
440    if include_config && config_path.exists() {
441        match fs::remove_file(config_path) {
442            Ok(_) => println!("✓ Deleted config at {:?}", config_path),
443            Err(e) => {
444                tracing::warn!("Failed to delete config: {}", e);
445                had_errors = true;
446            }
447        }
448    }
449
450    // Clear tokens from keyring (best effort, don't fail if keyring is unavailable)
451    match KeyringTokenStore::new() {
452        Ok(token_store) => {
453            if token_store.token_exists() {
454                if let Err(e) = token_store.clear_tokens() {
455                    tracing::warn!("Failed to clear tokens from keyring: {}", e);
456                    // Don't mark as error - keyring might not be available in CI
457                } else {
458                    println!("✓ Cleared tokens from keyring");
459                }
460            }
461        }
462        Err(e) => {
463            // Keyring not available (e.g., in CI), just log and continue
464            tracing::debug!("Keyring not available: {}", e);
465        }
466    }
467
468    if had_errors {
469        println!("Logs cleared with some warnings (see above).");
470    } else {
471        println!("All logs cleared.");
472    }
473
474    Ok(())
475}

510pub fn handle_logs(
511    file: PathBuf,
512    requests: bool,
513    responses: bool,
514    method: Option<String>,
515    tail: bool,
516    lines: Option<usize>,
517) -> Result<()> {
518    if !file.exists() {
519        return Err(anyhow::anyhow!("Log file {:?} not found", file));
520    }
521
522    let contents = fs::read_to_string(&file)?;
523    let all_lines: Vec<&str> = contents.lines().collect();
524
525    let filtered_lines: Vec<&str> = all_lines
526        .iter()
527        .filter(|line| {
528            if let Ok(json) = serde_json::from_str::<serde_json::Value>(line) {
529                // Filter by direction
530                if requests && json.get("direction") == Some(&serde_json::json!("response")) {
531                    return false;
532                }
533                if responses && json.get("direction") == Some(&serde_json::json!("request")) {
534                    return false;
535                }
536
537                // Filter by method
538                if let Some(ref m) = method {
539                    if let Some(content) = json.get("content").and_then(|c| c.as_str()) {
540                        if let Ok(rpc) = serde_json::from_str::<serde_json::Value>(content) {
541                            if rpc.get("method").and_then(|v| v.as_str()) != Some(m) {
542                                return false;
543                            }
544                        }
545                    }
546                }
547
548                true
549            } else {
550                false
551            }
552        })
553        .copied()
554        .collect();
555
556    let display_lines = if let Some(n) = lines {
557        filtered_lines.iter().rev().take(n).rev().copied().collect()
558    } else {
559        filtered_lines
560    };
561
562    for line in display_lines {
563        if let Ok(json) = serde_json::from_str::<serde_json::Value>(line) {
564            println!("{}", serde_json::to_string_pretty(&json)?);
565        }
566    }
567
568    if tail {
569        println!("\nTail mode not yet implemented.");
570    }
571
572    Ok(())
573}

575pub fn handle_doctor_jwt() -> Result<()> {
576    println!("JWT Token Information:");
577    println!();
578
579    // Try to initialize keyring token store
580    let token_store = match KeyringTokenStore::new() {
581        Ok(store) => {
582            tracing::debug!("Keyring token store initialized successfully");
583            store
584        }
585        Err(e) => {
586            tracing::error!("Failed to initialize keyring: {}", e);
587            println!("✗ Failed to initialize keyring: {}", e);
588            println!("\nThe keyring may not be available on this system.");
589            println!("Run 'km init' to configure authentication.");
590            return Ok(());
591        }
592    };
593
594    // Check if token exists
595    let exists = token_store.token_exists();
596    tracing::debug!("Token exists in keyring: {}", exists);
597
598    if !exists {
599        println!("✗ No JWT token found in keyring");
600        println!("\nRun 'km init' to authenticate and store a token.");
601
602        // Try to get raw password to debug
603        tracing::debug!("Attempting to read raw keyring entry for debugging...");
604        return Ok(());
605    }
606
607    // Load the token
608    tracing::debug!("Attempting to load access token from keyring...");
609    match token_store.load_access_token() {
610        Ok(jwt_token) => {
611            // Display token (truncated for security)
612            let token_display = if jwt_token.token.len() > 40 {
613                format!(
614                    "{}...{}",
615                    &jwt_token.token[..20],
616                    &jwt_token.token[jwt_token.token.len() - 20..]
617                )
618            } else {
619                jwt_token.token.clone()
620            };
621            println!("  Token: {}", token_display);
622
623            // Display expiration
624            use std::time::{SystemTime, UNIX_EPOCH};
625            let now = SystemTime::now()
626                .duration_since(UNIX_EPOCH)
627                .unwrap()
628                .as_secs();
629
630            let is_expired = AuthClient::is_token_expired(&jwt_token);
631            let expires_in = jwt_token.expires_at.saturating_sub(now);
632
633            let expires_at_str = chrono::DateTime::from_timestamp(jwt_token.expires_at as i64, 0)
634                .map(|dt| dt.format("%Y-%m-%d %H:%M:%S UTC").to_string())
635                .unwrap_or_else(|| "Invalid timestamp".to_string());
636
637            println!(
638                "  Expires At: {} (in {} seconds)",
639                expires_at_str, expires_in
640            );
641
642            if is_expired {
643                println!("  Status: ✗ EXPIRED");
644            } else {
645                println!("  Status: ✓ Valid");
646            }
647
648            println!();
649            println!("  Claims:");
650
651            if let Some(user_id) = &jwt_token.claims.user_id {
652                println!("    User ID: {}", user_id);
653            }
654
655            if let Some(tier) = &jwt_token.claims.tier {
656                println!("    Tier: {}", tier);
657            }
658
659            if let Some(sub) = &jwt_token.claims.sub {
660                println!("    Subject: {}", sub);
661            }
662
663            if let Some(iat) = jwt_token.claims.iat {
664                let iat_str = chrono::DateTime::from_timestamp(iat as i64, 0)
665                    .map(|dt| dt.format("%Y-%m-%d %H:%M:%S UTC").to_string())
666                    .unwrap_or_else(|| "Invalid timestamp".to_string());
667                println!("    Issued At: {}", iat_str);
668            }
669
670            println!();
671
672            if is_expired {
673                println!("⚠ Token is expired. Run 'km monitor' to automatically refresh it,");
674                println!("  or run 'km init' to re-authenticate.");
675            }
676
677            Ok(())
678        }
679        Err(e) => {
680            tracing::error!("Failed to load JWT token from keyring: {}", e);
681            println!("✗ Failed to load JWT token: {}", e);
682            println!("\nThe token may be corrupted. Run 'km init' to re-authenticate.");
683            Ok(())
684        }
685    }
