1use crate::auth::JwtToken;
2use anyhow::{Context, Result};
3use keyring::Entry;
4
5const SERVICE_NAME: &str = "ai.kilometers.km";
6const ACCESS_TOKEN_KEY: &str = "km-access-token";
7const REFRESH_TOKEN_KEY: &str = "km-refresh-token";
8
9/// Check if running in a CI environment where keyring access may not be available
10fn is_ci_environment() -> bool {
11    std::env::var("CI").is_ok() || std::env::var("GITHUB_ACTIONS").is_ok()
12}
13
14pub struct KeyringTokenStore {
15    access_token_entry: Entry,
16    refresh_token_entry: Entry,
17}
18
19impl KeyringTokenStore {
20    pub fn new() -> Result<Self> {
21        // In CI environments, keyring access may hang or fail
22        // Return an error that can be handled gracefully
23        if is_ci_environment() {
24            tracing::debug!("Running in CI environment, keyring operations will be skipped");
25            return Err(anyhow::anyhow!("Keyring not available in CI environment"));
26        }
27
28        let access_token_entry = Entry::new(SERVICE_NAME, ACCESS_TOKEN_KEY)
29            .context("Failed to create keyring entry for access token")?;
30
31        let refresh_token_entry = Entry::new(SERVICE_NAME, REFRESH_TOKEN_KEY)
32            .context("Failed to create keyring entry for refresh token")?;
33
34        Ok(Self {
35            access_token_entry,
36            refresh_token_entry,
37        })
38    }
39
40    pub fn save_tokens(&self, access_token: &JwtToken, refresh_token: Option<&str>) -> Result<()> {
41        // Serialize access token to JSON
42        let access_token_json =
43            serde_json::to_string(access_token).context("Failed to serialize access token")?;
44
45        // Save access token to keyring
46        self.access_token_entry
47            .set_password(&access_token_json)
48            .context("Failed to save access token to keyring")?;
49
50        tracing::debug!("Access token saved to keyring");
51
52        // Save refresh token if provided and non-empty
53        // Note: Linux and Windows keyring implementations reject empty strings
54        if let Some(refresh) = refresh_token {
55            if !refresh.is_empty() {
56                self.refresh_token_entry
57                    .set_password(refresh)
58                    .context("Failed to save refresh token to keyring")?;
59                tracing::debug!("Refresh token saved to keyring");
60            } else {
61                tracing::debug!("Skipping empty refresh token (not supported by keyring)");
62            }
63        }
64
65        Ok(())
66    }
67
68    pub fn load_access_token(&self) -> Result<JwtToken> {
69        let access_token_json = self
70            .access_token_entry
71            .get_password()
72            .context("Failed to retrieve access token from keyring")?;
73
74        let token: JwtToken = serde_json::from_str(&access_token_json)
75            .context("Failed to deserialize access token")?;
76
77        Ok(token)
78    }
79
80    #[allow(dead_code)]
81    pub fn load_refresh_token(&self) -> Result<Option<String>> {
82        match self.refresh_token_entry.get_password() {
83            Ok(refresh_token) => Ok(Some(refresh_token)),
84            Err(keyring::Error::NoEntry) => Ok(None),
85            Err(e) => Err(anyhow::anyhow!("Failed to retrieve refresh token: {}", e)),
86        }
87    }
88
89    pub fn clear_tokens(&self) -> Result<()> {
90        // Try to delete access token
91        match self.access_token_entry.delete_credential() {
92            Ok(_) => tracing::info!("Access token cleared from keyring"),
93            Err(keyring::Error::NoEntry) => {
94                tracing::debug!("No access token found in keyring to clear")
95            }
96            Err(e) => return Err(anyhow::anyhow!("Failed to clear access token: {}", e)),
97        }
98
99        // Try to delete refresh token
100        match self.refresh_token_entry.delete_credential() {
101            Ok(_) => tracing::info!("Refresh token cleared from keyring"),
102            Err(keyring::Error::NoEntry) => {
103                tracing::debug!("No refresh token found in keyring to clear")
104            }
105            Err(e) => return Err(anyhow::anyhow!("Failed to clear refresh token: {}", e)),
106        }
107
108        Ok(())
109    }
110
111    pub fn token_exists(&self) -> bool {
112        self.access_token_entry.get_password().is_ok()
113    }
114}
